        function useLocalStorage(key, initialValue) {
          const [storedValue, setStoredValue] = useState(() => {
            try {
              const item = window.localStorage.getItem(key);
              return item ? JSON.parse(item) : initialValue;
            } catch (error) { return initialValue; }
          });

          const setValue = (value) => {
            try {
              const valueToStore = value instanceof Function ? value(storedValue) : value;
              setStoredValue(valueToStore);
              window.localStorage.setItem(key, JSON.stringify(valueToStore));
            } catch (error) { console.error("Error guardando en memoria local", error); }
          };
          return [storedValue, setValue];
        }

          const estadisticas = useMemo(() => jornadasFiltradas.reduce((a, c) => ({ dias: a.dias + 1, total: a.total + c.totalGenerado }), { dias: 0, total: 0 }), [jornadasFiltradas]);

                {/* --- REGISTRO --- */}

                          { id: 'doble', label: 'Doble Turno', state: esDobleTurno, set: setEsDobleTurno, icon: "fa-dollar-sign", color: "text-emerald-500", monto: valoresActuales.BONO_DOBLE_TURNO },

                                <i className={`fa-solid ${item.icon} ${item.color} text-lg`}></i>
                              </div>

                                <span className="block font-bold text-slate-700">{item.label}</span>
